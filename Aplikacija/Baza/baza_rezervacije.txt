CREATE TABLE `rezervacije` (
  `idrezervacije` int unsigned NOT NULL AUTO_INCREMENT,
  `idturisteRez` int unsigned NOT NULL,
  `idvodicaRez` int unsigned NOT NULL,
  `brojOsoba` int unsigned NOT NULL,
  `datumIzvodjenja` datetime(1) DEFAULT NULL,
  `idTureRez` int unsigned DEFAULT NULL,
  PRIMARY KEY (`idrezervacije`),
  KEY `idTureRez_idx` (`idTureRez`),
  KEY `idTuristeRez_idx` (`idturisteRez`),
  KEY `idVodicaRez_idx` (`idvodicaRez`),
  CONSTRAINT `idTureRez` FOREIGN KEY (`idTureRez`) REFERENCES `ture` (`idture`),
  CONSTRAINT `idTuristeRez` FOREIGN KEY (`idturisteRez`) REFERENCES `turisti` (`idturiste`),
  CONSTRAINT `idVodicaRez` FOREIGN KEY (`idvodicaRez`) REFERENCES `vodici` (`idvodica`)
) ENGINE=InnoDB AUTO_INCREMENT=10 DEFAULT CHARSET=latin1

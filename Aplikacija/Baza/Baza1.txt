CREATE TABLE `korisnici` (
  `idkorisnika` int unsigned NOT NULL AUTO_INCREMENT,
  `username` varchar(45) NOT NULL,
  `password` varchar(45) DEFAULT NULL,
  `idturisteKor` int unsigned DEFAULT NULL,
  `idvodicaKor` int unsigned DEFAULT NULL,
  PRIMARY KEY (`idkorisnika`),
  UNIQUE KEY `idkorisnici_UNIQUE` (`idkorisnika`),
  UNIQUE KEY `username_UNIQUE` (`username`)
) ENGINE=InnoDB AUTO_INCREMENT=12 DEFAULT CHARSET=latin1


CREATE TABLE `turisti` (
  `idturiste` int unsigned NOT NULL AUTO_INCREMENT,
  `imeTuriste` varchar(45) NOT NULL,
  `prezimeTuriste` varchar(45) DEFAULT NULL,
  `pol` char(1) DEFAULT NULL,
  `datumRodjenja` date DEFAULT NULL,
  `email` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`idturiste`),
  UNIQUE KEY `idturiste_UNIQUE` (`idturiste`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1

CREATE TABLE `vodici` (
  `idvodica` int unsigned NOT NULL AUTO_INCREMENT,
  `imeVodica` varchar(45) DEFAULT NULL,
  `prezimeVodica` varchar(45) DEFAULT NULL,
  `pol` char(1) DEFAULT NULL,
  `brojTelefona` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`idvodica`),
  UNIQUE KEY `idvodica_UNIQUE` (`idvodica`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=latin1

CREATE TABLE `rezervacije` (
  `idrezervacije` int(10) unsigned NOT NULL auto_increment,
  `idturisteRez` int(10) unsigned NOT NULL,
  `idvodicaRez` int(10) unsigned NOT NULL,
  `brojOsoba` int(10) unsigned NOT NULL,
  `datumIzvodjenja` datetime(1) DEFAULT NULL,
  PRIMARY KEY (`idrezervacije`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1


CREATE TABLE `ture` (
  `idture` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `nazivTure` varchar(45) DEFAULT NULL,
  `trajanjeTure` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`idture`),
  UNIQUE KEY `idture_UNIQUE` (`idture`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1


CREATE TABLE `znamenitosti` (
  `idznamenitosti` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `nazivZnamenitosti` varchar(45) DEFAULT NULL,
  `lokacijaZnamenitosti` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`idznamenitosti`),
  UNIQUE KEY `idznamenitosti_UNIQUE` (`idznamenitosti`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1

CREATE TABLE `ocenjivanjevodica` (
  `idocenjivanjevodica` int unsigned NOT NULL AUTO_INCREMENT,
  `idvodicaOcenjivanje` int unsigned DEFAULT NULL,
  `ocena` int unsigned DEFAULT NULL,
  PRIMARY KEY (`idocenjivanjevodica`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

 CREATE TABLE `kviz` (
  `idkviz` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `nazivkviza` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`idkviz`));

CREATE TABLE `pitanje` (
  `idpitanje` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `tekstpitanja` VARCHAR(1000) NOT NULL,
  `kvizid` INT UNSIGNED NOT NULL,
  PRIMARY KEY (`idpitanje`),
  INDEX `kvizid_idx` (`kvizid`),
  CONSTRAINT `kvizid`
    FOREIGN KEY (`kvizid`)
    REFERENCES `kviz` (`idkviz`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

CREATE TABLE `odgovor` (
  `idodgovor` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `tekstodg` VARCHAR(1000) NOT NULL,
  `pitanjeid` INT UNSIGNED NOT NULL,
  PRIMARY KEY (`idodgovor`),
  INDEX `pitanjeid_idx` (`pitanjeid`),
  CONSTRAINT `pitanjeid`
    FOREIGN KEY (`pitanjeid`)
    REFERENCES `pitanje` (`idpitanje`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
